106 function softscope(@nospecialize ex)
107 if ex isa Expr
108 h = ex.head
109 if h === :toplevel
110 ex′ = Expr(h)
111 map!(softscope, resize!(ex′.args, length(ex.args)), ex.args)
112 return ex′
113 elseif h in (:meta, :import, :using, :export, :module, :error, :incomplete, :thunk)
114 return ex
115 elseif h === :global && all(x->isa(x, Symbol), ex.args)
116 return ex
117 else
118 return Expr(:block, Expr(:softscope, true), ex)
119 end
120 end
121 return ex
122 end

553 function hist_from_file(hp::REPLHistoryProvider, path::String)
554 getline(lines, i) = i > length(lines) ? "" : lines[i]
555 file_lines = readlines(path)
556 countlines = 0
557 while true
558 # First parse the metadata that starts with '#' in particular the REPL mode
559 countlines += 1
560 line = getline(file_lines, countlines)
561 mode = :julia
562 isempty(line) && break
563 line[1] != '#' &&
564 error(invalid_history_message(path), repr(line[1]), " at line ", countlines)
565 while !isempty(line)
566 startswith(line, '#') || break
567 if startswith(line, "# mode: ")
568 mode = Symbol(SubString(line, 9))
569 end
570 countlines += 1
571 line = getline(file_lines, countlines)
572 end
573 isempty(line) && break
574
575 # Now parse the code for the current REPL mode
576 line[1] == ' ' &&
577 error(munged_history_message(path), countlines)
578 line[1] != '\t' &&
579 error(invalid_history_message(path), repr(line[1]), " at line ", countlines)
580 lines = String[]
581 while !isempty(line)
582 push!(lines, chomp(SubString(line, 2)))
583 next_line = getline(file_lines, countlines+1)
584 isempty(next_line) && break
585 first(next_line) == ' ' && error(munged_history_message(path), countlines)
586 # A line not starting with a tab means we are done with code for this entry
587 first(next_line) != '\t' && break
588 countlines += 1
589 line = getline(file_lines, countlines)
590 end
591 push!(hp.modes, mode)
592 push!(hp.history, join(lines, '\n'))
593 end
594 hp.start_idx = length(hp.history)

705 if m === :ok
706 LineEdit.move_input_end(s)
707 return LineEdit.refresh_line(s)
708 elseif m === :skip
709 return history_next(s, hist, num+1, save_idx)
710 else
711 return Terminals.beep(s)
712 end
713 end

722 function history_move_prefix(s::LineEdit.PrefixSearchState,
723 hist::REPLHistoryProvider,
724 prefix::AbstractString,
725 backwards::Bool,
726 cur_idx::Int = hist.cur_idx)
727 cur_response = String(take!(copy(LineEdit.buffer(s))))
728 # when searching forward, start at last_idx
729 if !backwards && hist.last_idx > 0
730 cur_idx = hist.last_idx
731 end
732 hist.last_idx = -1
733 max_idx = length(hist.history)+1
734 idxs = backwards ? ((cur_idx-1):-1:1) : ((cur_idx+1):1:max_idx)
735 for idx in idxs
736 if (idx == max_idx) || (startswith(hist.history[idx], prefix) && (hist.history[idx] != cur_response || hist.modes[idx] != LineEdit.mode(s)))
737 m = history_move(s, hist, idx)
738 if m === :ok
739 if idx == max_idx
740 # on resuming the in-progress edit, leave the cursor where the user last had it
741 elseif isempty(prefix)
742 # on empty prefix search, move cursor to the end
743 LineEdit.move_input_end(s)
744 else
745 # otherwise, keep cursor at the prefix position as a visual cue
746 seek(LineEdit.buffer(s), sizeof(prefix))
747 end
748 LineEdit.refresh_line(s)
749 return :ok
750 elseif m === :skip
751 return history_move_prefix(s,hist,prefix,backwards,idx)
752 end
753 end
754 end
755 Terminals.beep(s)
756 nothing
757 end

759 history_move_prefix(s, hist, prefix, false)

761 history_move_prefix(s, hist, prefix, true)

763 function history_search(hist::REPLHistoryProvider, query_buffer::IOBuffer, response_buffer::IOBuffer,
764 backwards::Bool=false, skip_current::Bool=false)
765
766 qpos = position(query_buffer)
767 qpos > 0 || return true
768 searchdata = beforecursor(query_buffer)
769 response_str = String(take!(copy(response_buffer)))
770
771 # Alright, first try to see if the current match still works
772 a = position(response_buffer) + 1 # position is zero-indexed
773 # FIXME: I'm pretty sure this is broken since it uses an index
774 # into the search data to index into the response string
775 b = a + sizeof(searchdata)
776 b = b ≤ ncodeunits(response_str) ? prevind(response_str, b) : b-1
777 b = min(lastindex(response_str), b) # ensure that b is valid
778
779 searchstart = backwards ? b : a
780 if searchdata == response_str[a:b]
781 if skip_current
782 searchstart = backwards ? prevind(response_str, b) : nextind(response_str, a)
783 else
784 return true
785 end
786 end
787
788 # Start searching
789 # First the current response buffer
790 if 1 <= searchstart <= lastindex(response_str)
791 match = backwards ? findprev(searchdata, response_str, searchstart) :
792 findnext(searchdata, response_str, searchstart)
793 if match !== nothing
794 seek(response_buffer, first(match) - 1)
795 return true
796 end
797 end
798
799 # Now search all the other buffers
800 idxs = backwards ? ((hist.cur_idx-1):-1:1) : ((hist.cur_idx+1):1:length(hist.history))
801 for idx in idxs
802 h = hist.history[idx]
803 match = backwards ? findlast(searchdata, h) : findfirst(searchdata, h)
804 if match !== nothing && h != response_str && haskey(hist.mode_mapping, hist.modes[idx])
805 truncate(response_buffer, 0)
806 write(response_buffer, h)
807 seek(response_buffer, first(match) - 1)
808 hist.cur_idx = idx
809 return true
810 end
811 end
812
813 return false
814 end

821 nothing

911 function setup_interface(

917 # type of `Union{Dict,Vector{<:Dict}}` (see #28808).
918 # This function is however important to precompile for REPL startup time, therefore,

924 # This function returns the main interface that describes the REPL
925 # functionality, it is called internally by functions that setup a

1035 ';' => function (s::MIState,o...)
1036 if isempty(s) || position(LineEdit.buffer(s)) == 0
1037 buf = copy(LineEdit.buffer(s))
1038 transition(s, shell_mode) do
1039 LineEdit.state(s, shell_mode).input_buffer = buf
1040 end
1041 else

1043 end
1044 end,
1045 '?' => function (s::MIState,o...)
1046 if isempty(s) || position(LineEdit.buffer(s)) == 0
1047 buf = copy(LineEdit.buffer(s))
1048 transition(s, help_mode) do
1049 LineEdit.state(s, help_mode).input_buffer = buf
1050 end
1051 else
1052 edit_insert(s, '?')
1053 end
1054 end,
1055
1056 # Bracketed Paste Mode
1057 "\e[200~" => (s::MIState,o...)->begin
1058 input = LineEdit.bracketed_paste(s) # read directly from s until reaching the end-bracketed-paste marker
1059 sbuffer = LineEdit.buffer(s)
1060 curspos = position(sbuffer)
1061 seek(sbuffer, 0)
1062 shouldeval = (bytesavailable(sbuffer) == curspos && !occursin(UInt8('\n'), sbuffer))
1063 seek(sbuffer, curspos)
1064 if curspos == 0
1065 # if pasting at the beginning, strip leading whitespace
1066 input = lstrip(input)
1067 end
1068 if !shouldeval
1069 # when pasting in the middle of input, just paste in place
1070 # don't try to execute all the WIP, since that's rather confusing
1071 # and is often ill-defined how it should behave
1072 edit_insert(s, input)
1073 return
1074 end
1075 LineEdit.push_undo(s)
1076 edit_insert(sbuffer, input)
1077 input = String(take!(sbuffer))
1078 oldpos = firstindex(input)
1079 firstline = true
1080 isprompt_paste = false
1081 curr_prompt_len = 0
1082 pasting_help = false
1083
1084 while oldpos <= lastindex(input) # loop until all lines have been executed
1085 if JL_PROMPT_PASTE[]
1086 # Check if the next statement starts with a prompt i.e. "julia> ", in that case
1087 # skip it. But first skip whitespace unless pasting in a docstring which may have
1088 # indented prompt examples that we don't want to execute
1089 while input[oldpos] in (pasting_help ? ('\n') : ('\n', ' ', '\t'))
1090 oldpos = nextind(input, oldpos)
1091 oldpos >= sizeof(input) && return
1092 end
1093 # Check if input line starts with "julia> ", remove it if we are in prompt paste mode
1094 if (firstline || isprompt_paste) && startswith(SubString(input, oldpos), JULIA_PROMPT)
1095 isprompt_paste = true
1096 oldpos += jl_prompt_len
1097 curr_prompt_len = jl_prompt_len
1098 transition(s, julia_prompt)
1099 pasting_help = false
1100 # Check if input line starts with "pkg> " or "(...) pkg> ", remove it if we are in prompt paste mode and switch mode
1101 elseif (firstline || isprompt_paste) && startswith(SubString(input, oldpos), pkg_prompt_regex)
1102 detected_pkg_prompt = match(pkg_prompt_regex, SubString(input, oldpos)).match
1103 isprompt_paste = true
1104 curr_prompt_len = sizeof(detected_pkg_prompt)
1105 oldpos += curr_prompt_len
1106 Base.active_repl.interface.modes[1].keymap_dict[']'](s, o...)
1107 pasting_help = false
1108 # Check if input line starts with "shell> ", remove it if we are in prompt paste mode and switch mode
1109 elseif (firstline || isprompt_paste) && startswith(SubString(input, oldpos), SHELL_PROMPT)
1110 isprompt_paste = true
1111 oldpos += shell_prompt_len
1112 curr_prompt_len = shell_prompt_len
1113 transition(s, shell_mode)
1114 pasting_help = false
1115 # Check if input line starts with "help?> ", remove it if we are in prompt paste mode and switch mode
1116 elseif (firstline || isprompt_paste) && startswith(SubString(input, oldpos), HELP_PROMPT)
1117 isprompt_paste = true
1118 oldpos += help_prompt_len
1119 curr_prompt_len = help_prompt_len
1120 transition(s, help_mode)
1121 pasting_help = true
1122 # If we are prompt pasting and current statement does not begin with a mode prefix, skip to next line
1123 elseif isprompt_paste
1124 while input[oldpos] != '\n'
1125 oldpos = nextind(input, oldpos)
1126 oldpos >= sizeof(input) && return
1127 end
1128 continue
1129 end
1130 end
1131 dump_tail = false
1132 nl_pos = findfirst('\n', input[oldpos:end])
1133 if s.current_mode == julia_prompt
1134 ast, pos = Meta.parse(input, oldpos, raise=false, depwarn=false)
1135 if (isa(ast, Expr) && (ast.head === :error || ast.head === :incomplete)) ||
1136 (pos > ncodeunits(input) && !endswith(input, '\n'))
1137 # remaining text is incomplete (an error, or parser ran to the end but didn't stop with a newline):
1138 # Insert all the remaining text as one line (might be empty)
1139 dump_tail = true
1140 end
1141 elseif isnothing(nl_pos) # no newline at end, so just dump the tail into the prompt and don't execute
1142 dump_tail = true
1143 elseif s.current_mode == shell_mode # handle multiline shell commands
1144 lines = split(input[oldpos:end], '\n')
1145 pos = oldpos + sizeof(lines[1]) + 1
1146 if length(lines) > 1
1147 for line in lines[2:end]
1148 # to be recognized as a multiline shell command, the lines must be indented to the
1149 # same prompt position
1150 if !startswith(line, ' '^curr_prompt_len)
1151 break
1152 end
1153 pos += sizeof(line) + 1
1154 end
1155 end
1156 else
1157 pos = oldpos + nl_pos
1158 end
1159 if dump_tail
1160 tail = input[oldpos:end]
1161 if !firstline
1162 # strip leading whitespace, but only if it was the result of executing something
1163 # (avoids modifying the user's current leading wip line)
1164 tail = lstrip(tail)
1165 end
1166 if isprompt_paste # remove indentation spaces corresponding to the prompt
1167 tail = replace(tail, r"^"m * ' '^curr_prompt_len => "")
1168 end
1169 LineEdit.replace_line(s, tail, true)
1170 LineEdit.refresh_line(s)
1171 break
1172 end
1173 # get the line and strip leading and trailing whitespace
1174 line = strip(input[oldpos:prevind(input, pos)])
1175 if !isempty(line)
1176 if isprompt_paste # remove indentation spaces corresponding to the prompt
1177 line = replace(line, r"^"m * ' '^curr_prompt_len => "")
1178 end
1179 # put the line on the screen and history
1180 LineEdit.replace_line(s, line)
1181 LineEdit.commit_line(s)
1182 # execute the statement
1183 terminal = LineEdit.terminal(s) # This is slightly ugly but ok for now
1184 raw!(terminal, false) && disable_bracketed_paste(terminal)
1185 LineEdit.mode(s).on_done(s, LineEdit.buffer(s), true)
1186 raw!(terminal, true) && enable_bracketed_paste(terminal)
1187 LineEdit.push_undo(s) # when the last line is incomplete
1188 end
1189 oldpos = pos
1190 firstline = false
1191 end
1192 end,
1193
1194 # Open the editor at the location of a stackframe or method
1195 # This is accessing a contextual variable that gets set in
1196 # the show_backtrace and show_method_table functions.
1197 "^Q" => (s::MIState, o...) -> begin
1198 linfos = repl.last_shown_line_infos
1199 str = String(take!(LineEdit.buffer(s)))
1200 n = tryparse(Int, str)
1201 n === nothing && @goto writeback
1202 if n <= 0 || n > length(linfos) || startswith(linfos[n][1], "REPL[")
1203 @goto writeback
1204 end
1205 try
1206 InteractiveUtils.edit(linfos[n][1], linfos[n][2])
1207 catch ex
1208 ex isa ProcessFailedException || ex isa Base.IOError || ex isa SystemError || rethrow()
1209 @info "edit failed" _exception=ex
1210 end
1211 LineEdit.refresh_line(s)
1212 return
1213 @label writeback
1214 write(LineEdit.buffer(s), str)
1215 return
1216 end,

1276 let matchend = Dict("\"" => r"\"", "\"\"\"" => r"\"\"\"", "'" => r"'",
1277 "`" => r"`", "```" => r"```", "#" => r"$"m, "#=" => r"=#|#=")
1278 global _rm_strings_and_comments
1279 function _rm_strings_and_comments(code::Union{String,SubString{String}})
1280 buf = IOBuffer(sizehint = sizeof(code))
1281 pos = 1
1282 while true
1283 i = findnext(r"\"(?!\"\")|\"\"\"|'|`(?!``)|```|#(?!=)|#=", code, pos)
1284 isnothing(i) && break
1285 match = SubString(code, i)
1286 j = findnext(matchend[match]::Regex, code, nextind(code, last(i)))
1287 if match == "#=" # possibly nested
1288 nested = 1
1289 while j !== nothing
1290 nested += SubString(code, j) == "#=" ? +1 : -1
1291 iszero(nested) && break
1292 j = findnext(r"=#|#=", code, nextind(code, last(j)))
1293 end
1294 elseif match[1] != '#' # quote match: check non-escaped
1295 while j !== nothing
1296 notbackslash = findprev(!=('\\'), code, prevind(code, first(j)))::Int
1297 isodd(first(j) - notbackslash) && break # not escaped
1298 j = findnext(matchend[match]::Regex, code, nextind(code, first(j)))
